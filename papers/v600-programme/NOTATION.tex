% Shared notation for the V_600 programme (Papers 1–5).
% Each paper \input{NOTATION.tex} so symbols are identical across the set.

% =============================================================
% Groups and finite structures
% =============================================================
\newcommand{\twoI}{2I}                            % binary icosahedral group, |2I| = 120
\newcommand{\Vsixhundred}{V_{600}}                % 600-cell vertex set = 2I
\newcommand{\Dicfive}{\mathrm{Dic}_5}             % binary dihedral group, order 20
\newcommand{\Vtwentyfour}{V_{24}}                 % 24-cell sublattice, |V_24| = 24
\newcommand{\Esix}[1]{E_8}                        % E_8 lattice (reserved)
\newcommand{\WHfour}{W(H_4)}                      % H_4 Coxeter group
\newcommand{\Sptwo}{\mathrm{Sp}(1)}               % unit quaternion group

% =============================================================
% Cycle structure
% =============================================================
\newcommand{\Ttau}{T_\tau}                        % T_tau cycle generator
\newcommand{\cyc}[1]{\mathcal{C}_{#1}}            % cycle indexed by #1
\newcommand{\Kclass}[1]{P_{#1}}                   % K-class projector at K = #1
\newcommand{\Kmult}{\{72{:}1,\,0{:}1,\,52{:}5,\,20{:}5\}}  % K-multiset

% =============================================================
% Galois twist
% =============================================================
\newcommand{\sigmaG}{\sigma}                      % σ-Galois twist √5 ↦ -√5
\newcommand{\sigmafix}{\sigmaG\text{-fix}}        % σ-fixed
\newcommand{\sigmamob}{\sigmaG\text{-mob}}        % σ-mobile
\newcommand{\tausigma}{\tau_\sigmaG}              % canonical involution τ_σ

% =============================================================
% Trace metric and inner product on icosian ring
% =============================================================
\newcommand{\trinner}[2]{\langle #1,\, #2\rangle_{\mathrm{tr}}}  % trace inner product

% =============================================================
% Operators on cycle space
% =============================================================
\newcommand{\Hhat}{\hat{H}}                       % expansion operator on C
\newcommand{\Chat}{\hat{C}}                       % clustering operator on C
\newcommand{\Itwelve}{I_{12}}                     % identity on cycle space

% =============================================================
% Physical quantities
% =============================================================
\newcommand{\HZ}{H(z)}                            % Hubble parameter
\newcommand{\Hzero}{H_0}                          % Hubble constant
\newcommand{\sigeight}{\sigma_8}                  % matter clustering amplitude
\newcommand{\TCMB}{T_{\mathrm{CMB}}}              % CMB temperature
\newcommand{\TH}{T_H}                             % Hawking temperature
\newcommand{\Eq}{E_q}                             % cascade quantum gap
\newcommand{\SBH}{S_{\mathrm{BH}}}                % Bekenstein-Hawking entropy
\newcommand{\ABH}{A_{\mathrm{BH}}}                % horizon area

% =============================================================

\newcommand{\factorH}{\tfrac{13}{12}}             % 1 + 1/12
